package org.terasology.game;

/**
 */
public class Game {

    private String name = "";
    private String seed = "";

    public void load(GameManifest manifest) {
        this.name = manifest.getTitle();
        this.seed = manifest.getSeed();
    }

    public String getName() {
        return name;
    }

    public String getSeed() {
        return seed;
    }
}
